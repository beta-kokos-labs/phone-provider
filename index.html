// Setting a cookie document.cookie = "username=JohnDoe; expires=Fri, 31 Dec 2023 23:59:59 GMT; path=/"; // Reading a cookie const cookies = document.cookie.split(';'); const getCookieValue = (name) => { const cookie = cookies.find(c => c.trim().startsWith(name + '=')); return cookie ? cookie.split('=')[1] : null; }; console.log(getCookieValue('username')); // Outputs: JohnDoe
